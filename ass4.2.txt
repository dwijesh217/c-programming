#include <stdio.h>

int main() {
   char input;

   printf("Enter a character: ");
   scanf("%c", &input);

   (input >= 'a' && input <= 'z' || input >= 'A' && input <= 'Z') ? printf("%c is an alphabet\n", input) : printf("%c is not an alphabet\n", input);

   return 0;
}

Enter a character: p123
