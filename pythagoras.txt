#include <stdio.h>
int main() {
    int A;
    printf("Enter the limit: ");
    scanf("%d", &A);
    int m, n, o;
    for (n = 1; n < A; n++) {
        for (m = n+1; m < A; m++) {
            o = m*m + n*n;
            if (o >= A*A) {
                break; 
            }
            if (o % 1 == 0) { 
                printf("%d, %d, %d\n", m*m-n*n, 2*m*n, o);
            }
        }
    }

    return 0;
}
