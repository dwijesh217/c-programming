#include <stdio.h>
#include <stdlib.h>
void sort_array(int arr[], int n) {
    int i, j, temp;
    for (i = 0; i < n-1; i++) {
        for (j = 0; j < n-i-1; j++) {
            if (arr[j] > arr[j+1]) {
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}
int main() {
    int n, i, sum = 0;
    float mean;
    int mode, mode_count = 0, curr_count = 0;
    int median, mid;
    printf("Enter the number of elements: ");
    scanf("%d", &n);
    int *arr = (int *)malloc(n * sizeof(int));
    for (i = 0; i < n; i++) {
        printf("Enter element %d: ", i+1);
        scanf("%d", &arr[i]);
        sum += arr[i];
    }
    mean = (float)sum / n;
    sort_array(arr, n);
    mid = n/2;
    if (n % 2 == 0) {
        median = (arr[mid-1] + arr[mid]) / 2;
    } else {
        median = arr[mid];
    }
    for (i = 0; i < n; i++) {
        if (i == 0 || arr[i] != arr[i-1]) {
            curr_count = 1;
        } else {
            curr_count++;
        }
        if (curr_count > mode_count) {
            mode_count = curr_count;
            mode = arr[i];
        }
    }
    printf("Mean: %.2f\n", mean);
    printf("Median: %d\n", median);
    printf("Mode: %d\n", mode);
    free(arr);
    return 0;
}
